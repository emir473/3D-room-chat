// Initialisierung der Szene const scene = new THREE.Scene(); scene.background = new THREE.Color(0x333333); // Ein dunkler Hintergrund für den Raum // 1. Kamera-Position (1:1 wie im Bild: schräg von oben auf den Tisch) const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000); camera.position.set(0, 4.5, 6); camera.lookAt(0, 0, 0); // 2. Renderer-Setup (Wichtig für Schärfe und Schatten!) const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); renderer.setPixelRatio(window.devicePixelRatio); renderer.shadowMap.enabled = true; // Schatten aktivieren für Tiefe renderer.shadowMap.type = THREE.PCFSoftShadowMap; document.body.appendChild(renderer.domElement); // Globales Licht für eine weiche Ausleuchtung const ambientLight = new THREE.AmbientLight(0xffffff, 0.5); scene.add(ambientLight); // Der Room-Setup Code function setupRoom() { // 1. Boden const floorMat = new THREE.MeshStandardMaterial({color: 0x5c4033, roughness: 0.7}); const floor = new THREE.Mesh(new THREE.BoxGeometry(12, 0.2, 10), floorMat); floor.position.y = -0.1; floor.receiveShadow = true; // Boden empfängt Schatten scene.add(floor); // 2. Roter Teppich const carpetMat = new THREE.MeshStandardMaterial({color: 0x990000, roughness: 0.9}); const carpet = new THREE.Mesh(new THREE.BoxGeometry(5.5, 0.02, 3.5), carpetMat); carpet.position.set(0, 0.01, 0); scene.add(carpet); // 3. Wand hinten const wallMat = new THREE.MeshStandardMaterial({color: 0xe67e22, roughness: 0.6}); const backWall = new THREE.Mesh(new THREE.BoxGeometry(12, 6, 0.2), wallMat); backWall.position.set(0, 3, -5); scene.add(backWall); // 4. Fenster const skyMat = new THREE.MeshBasicMaterial({color: 0x5dade2}); const sky = new THREE.Mesh(new THREE.PlaneGeometry(3, 2), skyMat); sky.position.set(0, 3.5, -4.95); scene.add(sky); const windowGroup = new THREE.Group(); windowGroup.position.set(0, 3.5, -4.9); const glassMat = new THREE.MeshStandardMaterial({color: 0xffffff, transparent: true, opacity: 0.3, roughness: 0.1}); const glass = new THREE.Mesh(new THREE.PlaneGeometry(3, 2), glassMat); windowGroup.add(glass); const frameMat = new THREE.MeshStandardMaterial({color: 0xffffff}); const frameGeo = new THREE.BoxGeometry(0.1, 2.1, 0.15); const borderLeft = new THREE.Mesh(frameGeo, frameMat); borderLeft.position.set(-1.5, 0, 0); const borderRight = new THREE.Mesh(frameGeo, frameMat); borderRight.position.set(1.5, 0, 0); windowGroup.add(borderLeft, borderRight); scene.add(windowGroup); // 5. Deckenlampe const lampGroup = new THREE.Group(); lampGroup.position.set(0, 6, 0); const wire = new THREE.Mesh(new THREE.CylinderGeometry(0.02, 0.02, 1.8), new THREE.MeshBasicMaterial({color: 0x111111})); wire.position.y = -0.9; const bulb = new THREE.Mesh(new THREE.SphereGeometry(0.15, 16, 16), new THREE.MeshBasicMaterial({color: 0xfff2cc})); bulb.position.y = -1.8; lampGroup.add(wire, bulb); scene.add(lampGroup); // 6. Lichtquellen const spotLight = new THREE.SpotLight(0xfff2cc, 3.5, 10, Math.PI / 4, 0.5, 1); spotLight.position.set(0, 4.2, 0); spotLight.target.position.set(0, 0, 0); spotLight.castShadow = true; // Spotlicht wirft Schatten für 1:1 Look scene.add(spotLight); scene.add(spotLight.target); // 7. Tisch const table = new THREE.Mesh(new THREE.BoxGeometry(3.5, 0.6, 1.8), new THREE.MeshStandardMaterial({color: 0x4a2c11, roughness: 0.5})); table.position.set(0, 0.3, 0); table.castShadow = true; // Tisch wirft Schatten auf Teppich scene.add(table); } // Aufruf der Raum-Erstellung setupRoom(); // Animations-Schleife function animate() { requestAnimationFrame(animate); renderer.render(scene, camera); } animate();
